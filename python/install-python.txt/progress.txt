git clone https://github.com/danrev21/pyenv.git
cd devops_lab
pip install -r requirements.txt

pyenv install -l # list all available versions
pyenv versions # list installed versions
pyenv install 3.9.5

pyenv virtualenv 3.9.5 myenv
pyenv local myenv
cat .python-version # myenv
python -V # Python 3.9.5
pyenv activate myenv
python -V # Python 3.8.11
pyenv deactivate 3.8.11
python -V # Python 3.9.5
pyenv uninstall myenv
pyenv versions
pyenv local 3.9.5 # switch python version

pyenv global 3.9.5

pyenv uninstall myenv
pyenv local --unset

pip install requests  # The "requests" library is a popular Python
		      # package for making HTTP requests and 
		      # working with APIs.
pip freeze            # list installed libraries
