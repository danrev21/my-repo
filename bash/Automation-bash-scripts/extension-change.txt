chext() {
  old_ext="$1"
  new_ext="$2"
  for file in *.${old_ext}; do
    mv -v -- "$file" "$(echo "$file" | sed "s/\.${old_ext}$/.${new_ext}/")"
  done
}

Usage:

chext [old_extension] [new_extension]
